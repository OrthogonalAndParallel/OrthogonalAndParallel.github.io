环境变量

~/.zshrc

export NVM_DIR="$([ -z "${XDG_CONFIG_HOME-}" ] && printf %s "${HOME}/.nvm" || printf %s "${XDG_CONFIG_HOME}/nvm")"
   [ -s "$NVM_DIR/nvm.sh" ] && \. "$NVM_DIR/nvm.sh"  # This loads nvm
   [ -s "$NVM_DIR/bash_completion" ] && \. "$NVM_DIR/bash_completion"  # This loads nvm bash_completion
export HOMEBREW_AUTO_UPDATE_SECS=86400
export BREW_HOME=usr/local
export M2_HOME=/Users/jinchen/Documents/program/apache-maven-3.8.8
export JAVA_HOME=/Library/Java/JavaVirtualMachines/jdk-21.jdk/Contents/Home
export CXF_HOME=/Users/jinchen/Documents/program/apache-cxf-3.5.9
export UV_HOME=/Users/jinchen/.local
export PATH=$HOME/.pyenv/bin:$UV_HOME/bin:$CXF_HOME/bin:$JAVA_HOME/bin:$M2_HOME/bin:$BREW_HOME/bin:$HOME/.cargo/bin:$PATH
alias java8='export JAVA_HOME=/Library/Java/JavaVirtualMachines/jdk-1.8.jdk/Contents/Home; export PATH=$JAVA_HOME/bin:$PATH'
alias java17='export JAVA_HOME=/Library/Java/JavaVirtualMachines/jdk-17.jdk/Contents/Home; export PATH=$JAVA_HOME/bin:$PATH'
alias java21='export JAVA_HOME=/Library/Java/JavaVirtualMachines/jdk-21.jdk/Contents/Home; export PATH=$JAVA_HOME/bin:$PATH'
